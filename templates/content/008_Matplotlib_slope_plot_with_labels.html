countries = ["United Kingdom", "France", "Germany", "Belgium", "Italy", "Spain", "Denmark"]

df_plot = (df
 .select("Year", Country=pl.col("Entity"), Value=pl.col("Share of people who are happy (Eurobarometer 2017)"))
 .filter(pl.col("Year").is_in([1985, 2015]))
 .filter(pl.col("Country").is_in(countries))
)

ax = plt.figure(figsize=(4,8))

def units(x, pos):
    return f"{x:0.0f}%"

for i in countries:
    df = df_plot.filter(pl.col("Country") == i)
    plt.plot("Year", "Value", data=df, marker="o")
    old_happy = None
    for x, y in zip(df["Year"], df["Value"]):
        if old_happy is not None:
            happy_diff = (y - old_happy) / old_happy * 100
            y_format = f"{happy_diff:.1f}%"
            if happy_diff > 1:
                arrow = "↑"
                color = "black"
            else:
                arrow = "↓"
                color = "#d75f02"
            plt.gca().annotate(f"{i} {y_format} {arrow}", xy=(x, y), xytext=(20, -2), 
                               textcoords='offset points', 
                               ha="left", va="bottom",
                               color=color)
        old_happy = y

plt.figtext(0, 0.92, 
            "Share of people who say they are happy", 
            fontsize=18, ha="left",
            fontdict={'fontname': 'Georgia'})

plt.figtext(0, 0.05, 
            "Data source: European Union - (https://europa.eu/eurobarometer)",
            fontsize=7, ha='left', 
            fontdict={"color": "grey"})

plt.gca().yaxis.set_major_formatter(units)
plt.gca().spines["top"].set_visible(False)
plt.ylim(60, 95)
plt.xticks([1985, 2015])

plt.show()

    oldest = None
    for x, y in zip(df["Year"], df["Value"]):
        y_format = f"{y:0.0f}%"
        if oldest is None:
            if round(y) in y_list_left:
                plt.gca().annotate(f"{i} {y_format}", xy=(x, y), xytext=(-20, -16), 
                        textcoords='offset points', ha="right", va="bottom",

                plt.gca().annotate(f"{i} {y_format}", xy=(x, y), xytext=(-20, -2), 
                        textcoords='offset points', ha="right", va="bottom",
                        color=line_color)
                y_list_left.append(round(y))
            oldest = 1
        else:
            if round(y) in y_list_right:
