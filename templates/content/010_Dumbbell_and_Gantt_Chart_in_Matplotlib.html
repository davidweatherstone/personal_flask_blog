headers = ["Name", "Start", "End", "Party"]

def flatten_cols(df):
    cols=['_'.join(map(str, vals))
          for vals in df.columns.to_flat_index()]
    df.columns = headers
    return df

start_date = (df_raw
                .iloc[:,[2,3,4,8]]
                .pipe(flatten_cols)
                .assign(Name=lambda df_: df_["Name"].str.split("[", expand=True)[0],
                        Start=lambda df_: pd.to_datetime(df_["Start"]),
                        Party=lambda df_: df_["Party"].str.split("(", expand=True)[0])
                .loc[:,["Name", "Start", "Party"]]
                .drop_duplicates(subset="Name", keep="first")
        )

pms_list = list(start_date.loc[:,"Name"])

df_plot = (df_raw
.iloc[:,[2,3,4,8]]
.pipe(flatten_cols)
.assign(
    Name=lambda df_: df_["Name"].str.split("[", expand=True)[0],
    Start=lambda df_: pd.to_datetime(df_["Start"]),
    End=lambda df_: df_["End"].replace("Incumbent", pd.to_datetime("now").strftime("%d %B %Y"))
)
.assign(End=lambda df_: pd.to_datetime(df_["End"]))
.assign(Tenure=lambda df_: (df_["End"] - df_["Start"]).astype(str).str.split(" ", expand=True)[0].astype("int32"))
.drop_duplicates()
.groupby(["Name"], as_index=False).agg(Tenure=("Tenure","sum"))
.assign(Placeholder=0)
.melt(id_vars=["Name"], value_vars=["Tenure", "Placeholder"])
.merge(start_date, left_on="Name", right_on="Name", how="left", validate="m:1")
.sort_values(["Start", "value"])
)

    .assign(

    .drop_duplicates()
    .query("Start > 1945")

        value_name="Date")

fig, ax = plt.subplots(figsize=(6, 6))

party_colours = {
    "Whig": "#eca417",
    "Tory": "#bbbddc",
    "Conservative": "#6cb4e4",
    "Peelite": "#98bb9b",
    "Liberal": "#fed600",
    "Labour": "#dc2828"
}

# Iterate through prime ministers
for i, pm in enumerate(df_plot_timeline["Name"].unique()):
    df = df_plot_timeline[df_plot_timeline["Name"] == pm]
    color = party_colours[df["Party"].iloc[0]]
    for _, row in df.iterrows():
        if row["variable"] == "Start":
            start_date = pd.to_datetime(row["Date"])
        else:
            end_date = pd.to_datetime(row["Date"])
            # Plot the segment
            ax.plot([start_date, end_date], [i, i], color=color, linewidth=8, solid_capstyle="butt")
            # Add a gap between segments
            ax.plot([end_date, end_date], [i, i], color="white", linewidth=8, solid_capstyle="butt")
    
    # Plot on the plot as an annotation
    radius = 10
    ax.plot(19300, 15, 'o', ms=radius * 2, mec='yellow', mfc='none', mew=2)
    
    ax.annotate('45 days', xy=(19300, 15), xytext=(-100, 0),
        fontname="Arial", fontweight=100,
        textcoords='offset points',
        color='black', size='medium',
        arrowprops=dict(
            arrowstyle='simple,tail_width=0.01,head_width=0.4,head_length=0.4',
            facecolor='b', shrinkB=radius * 1.6)
   )
