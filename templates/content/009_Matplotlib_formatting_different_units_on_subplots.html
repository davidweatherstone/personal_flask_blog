plot_data = df.select(
    pl.col("Entity").alias("Milk Type"),
    pl.col("Land use of milks (m2)").alias("Land use"),
    pl.col("GHG emissions of milks (kg CO2eq)").alias("Greenhouse gas emissions"),
    pl.col("Water use of milks (L)").alias("Freshwater use"),
    Eutrophication="Eutrophication from milks (g PO43-eq)",

colors_dict = {"Dairy milk": "#7189b0",
               "Oat milk": "#c9a57b",
               "Soy milk": "#b576b1",
               "Almond milk": "#cd7384",
               "Rice milk": "#339d98"}

formatters_dict = {
    0: lambda x, pos: f"{x:.0f} m²",
    1: lambda x, pos: f"{x:.1f} kg",
    2: lambda x, pos: f"{x:.0f} L",
    3: lambda x, pos: f"{x:.1f} g"
}

fig, axes = plt.subplots(layout="tight", 
                         nrows=2, ncols=2,
                         figsize=(8,5))

cols = plot_data.columns[1:]

for idx, col in enumerate(cols):
    ax = axes.flat[idx]
    df = plot_data.sort(col)
    
    colors = [colors_dict[type] for type in df["Milk Type"]]
    ax.barh(df["Milk Type"], df[col], color=colors, height=0.8)

    for rect in ax.patches:
        bar_x = rect.xy[0] + rect.get_width()  # Center x-position
        bar_height = rect.get_height()  # bar height (value)
        y_pos = rect.xy[1] + (bar_height / 2) - 0.08  # Adjust offset as needed
        
        units = formatters_dict[idx](bar_height, None).split()[1]
        ax.text(bar_x * 1.02, y_pos, f"{str(round(bar_x,2))} {units}", size=8)
    
    ax.set_title(col, loc="left")
    ax.tick_params(axis="both", labelsize=8)
    ax.xaxis.set_major_locator(plt.MaxNLocator(5))
    # ax.xaxis.set_major_formatter(FuncFormatter(formatters_dict[idx])) # Not required due to loops below removing ticks and labels
    
    ax.spines["top"].set_visible(False)
    ax.spines["right"].set_visible(False)
    ax.spines["bottom"].set_visible(False)
    
    ax.grid(axis="x", linestyle="dotted")
    ax.set_axisbelow(True)
    ax.yaxis.get_major_ticks(False)
    
    # Removing X axis ticks/labels
    for tick in ax.xaxis.get_major_ticks():
        tick.tick1line.set_visible(False)
        # tick.tick2line.set_visible(False)
        tick.label1.set_visible(False)
        # tick.label2.set_visible(False)

    # Removing Y axis ticks/labels
    for tick in ax.yaxis.get_major_ticks():
        tick.tick1line.set_visible(False)
        # tick.tick2line.set_visible(False)
        # tick.label1.set_visible(False)
        # tick.label2.set_visible(False)

plt.figtext(0, 1.11, "Environmental footprints of dairy and plant-based milks",
            fontsize=18, ha="left",
            fontdict={'fontname': 'Georgia'})

plt.figtext(0, 1.01, 
            "Impacts are measured per liter of milk. These are based on a meta-analysis of food system impact\n" \
            "studies across the supply chain which includes land use change, on-farm production, processing,\n" \
            "transport, and packaging.",
            fontsize=10, ha="left", color="grey",
            fontdict={'fontname': 'Arial'})

plt.figtext(0, -0.04, 
            "Data source: Joseph Poor and Thomas Nemecek (2018)\n" \
            "OurWorldInData.org/environmental-impacts-of-food | CC BY",
            fontsize=10, ha="left", color="grey",
            fontdict={'fontname': 'Arial'})

plt.show()
